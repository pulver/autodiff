1007 template<typename RealType,size_t Order>
1008 template<typename RealType2,size_t Order2>
1009 dimension<RealType,Order>& dimension<RealType,Order>::operator*=(const dimension<RealType2,Order2>& cr)
1010 {
1011  const promote<RealType,RealType2> zero{0};
1012  if BOOST_AUTODIFF_IF_CONSTEXPR (Order <= Order2)
1013  for (size_t i=0, j=Order ; i<=Order ; ++i, --j)
1014  v[j] = std::inner_product(v.cbegin(), v.cend()-i, cr.v.crbegin()+i, zero);
1015  else
1016  {
1017  for (size_t i=0, j=Order ; i<=Order-Order2 ; ++i, --j)
1018  v[j] = std::inner_product(cr.v.cbegin(), cr.v.cend(), v.crbegin()+i, zero);
1019  for (size_t i=Order-Order2+1, j=Order2-1 ; i<=Order ; ++i, --j)
1020  v[j] = std::inner_product(cr.v.cbegin(), cr.v.cbegin()+(j+1), v.crbegin()+i, zero);
1021  }
1022  return *this;
1023 }
